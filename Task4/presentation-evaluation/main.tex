\begin{frame}
\frametitle{Evaluation Criterion}
\begin{itemize}
\item Characteristics of our Data
\begin{itemize}
\item Multiple Labels
\item Strongly Imbalanced
\item Positive Events are Rare
\end{itemize}
\item Primary Metric: \textbf{F1 Score}
\begin{itemize}
\item Balance between Precision and Recall
\item High precision only $\rightarrow$ Sparse predictions \\
\textit{Only predict true events when really confident.}
\item High recall only $\rightarrow$ Greedy predictions \\
\textit{Cover more true events, even if it means a higher false positive rate.}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Baseline Classifier}
\begin{minipage}[t]{0.49\textwidth}
\begin{itemize}
\item Why not the majority class classifier?
\begin{itemize}
\item Majority Event is no Event at all
\item No True or False Positives\\$\implies$ F1 Value is \textbf{Undefined}
\end{itemize}
\end{itemize}
\begin{itemize}
\item Class Frequency Classifier
\begin{itemize}
\item Class Frequency becomes Probability
\item For every sample, each class has a probability to be predicted.
\item Ignores the actual data in the sample.
\end{itemize}
\end{itemize}
\end{minipage}

\includegraphics[width=\linewidth]{freq.pdf}
\end{minipage}
\end{frame}
